  authDomain: "lyrics-a63b8.firebaseapp.com",

  databaseURL: "https://lyrics-a63b8-default-rtdb.europe-west1.firebasedatabase.app",

  projectId: "lyrics-a63b8"

};


firebase.initializeApp(firebaseConfig);

const db = firebase.database();


let lyrics = [];

let index = 0;


db.ref("lyrics").on("value", snap => {

  lyrics = snap.val() || [];

});


db.ref("index").on("value", snap => {

  index = snap.val() || 0;

  if (lyrics[index]) {

    document.getElementById("current").innerText =

      lyrics[index].line1 + " / " + lyrics[index].line2;

  }

});


function next() {

  if (index < lyrics.length - 1) {

    db.ref("index").set(index + 1);

  }

}


function prev() {

  if (index > 0) {

    db.ref("index").set(index - 1);

  }

}
